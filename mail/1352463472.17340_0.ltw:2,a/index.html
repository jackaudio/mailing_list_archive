    conf.env['WITH_OPUS'] = False
    if conf.check_cfg(package='opus', atleast_version='0.9.0' ,

        if conf.check_cc(header_name='opus/opus_custom.h', mandatory=False):

            conf.env['WITH_OPUS'] = True
